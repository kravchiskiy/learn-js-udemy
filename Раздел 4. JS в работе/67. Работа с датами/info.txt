const now = new Date();
//компоненты даты getters
now.getFullYear() //2024
now.getMonth()  //2
now.getDate() //день номер в нашем месяце.
now.getHours()
секунды и милисекунды

now.getDay() // номер дня недели. начинаются с воскресенья.  0 - вс. 6 - сб.

все эти методы возвращают дату относительно местного часового пояса. 
поэтому у этих методов есть аналог, прибавив UTC. getUTCHours()

getTimeZoneOffset() - разница воо времени. между часовым поясом и utc. разница в минутах.
getTime() - timestamp.
